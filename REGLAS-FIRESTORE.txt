----- INICIO REGLAS -----
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function estaAutenticado() {
      return request.auth != null;
    }

    function esAdmin() {
      return estaAutenticado() &&
        request.auth.token.email in [
          'distribuidoraenzo28@gmail.com',
          'lucasbrizuela2002@gmail.com'
        ];
    }

    // ── PRODUCTOS ────────────────────────────────────────────────────────
    // Lectura pública. Escritura solo admin, EXCEPTO: el checkout del
    // cliente VIP descuenta stock con transacciones desde el navegador,
    // y para productos con variantes actualiza 'variantes' + 'stock'
    // JUNTOS. Por eso el hasOnly debe incluir ambos campos.
    match /productos/{productoId} {
      allow read: if true;

      allow update: if esAdmin() ||
              (estaAutenticado() &&
              request.resource.data.diff(resource.data)
                .affectedKeys().hasOnly(['stock', 'variantes']));

      allow create, delete: if esAdmin();
    }

    // ── PEDIDOS ──────────────────────────────────────────────────────────
    // Crear: cualquiera (visitantes sin cuenta mandan presupuestos).
    // Leer: admin todo; el cliente solo SUS pedidos.
    match /pedidos/{pedidoId} {
      allow read: if esAdmin() ||
        (estaAutenticado() && resource.data.emailUsuario == request.auth.token.email);
      allow create: if true;
      allow update, delete: if esAdmin();
    }

    // ── SOLICITUDES DE ACCESO ────────────────────────────────────────────
    // Validación ajustada al formulario real de cuenta.html:
    // obligatorios = nombre, dni, telefono, negocio. El email es OPCIONAL
    // (puede venir vacío) — por eso NO se valida su tamaño.
    match /solicitudes_acceso/{solicitudId} {
        allow create: if request.resource.data.keys().hasAll(['nombre', 'dni', 'telefono', 'negocio', 'estado', 'fecha'])
                      && request.resource.data.nombre is string
                      && request.resource.data.dni is string
                      && request.resource.data.telefono is string
                      && request.resource.data.negocio is string
                      && request.resource.data.estado == 'pendiente'
                      && request.resource.data.nombre.size() > 1
                      && request.resource.data.nombre.size() < 200
                      && request.resource.data.dni.size() >= 7
                      && request.resource.data.dni.size() <= 8
                      && request.resource.data.telefono.size() > 5;
        allow read, update, delete: if esAdmin();
    }

    // ── USUARIOS PRE-APROBADOS (legacy) ──────────────────────────────────
    // Ningún código la usa hoy; se conserva para que el admin pueda
    // consultar/limpiar los datos viejos.
    match /usuarios_preaprobados/{email} {
      allow read, write: if esAdmin();
    }

    // ── PERFIL DE USUARIOS ───────────────────────────────────────────────
    match /usuarios/{userId} {
      allow read: if esAdmin() || (estaAutenticado() && request.auth.uid == userId);

      // El admin crea cuentas al aprobar; el usuario solo puede crear
      // su propio documento.
      allow create: if esAdmin() || (estaAutenticado() && request.auth.uid == userId);

      // ESCUDO: el usuario puede editar su perfil (nombre, teléfono,
      // dirección, foto, etc.) pero NO puede cambiar los campos
      // sensibles: aprobado (VIP), dni, cuit, email.
      allow update: if esAdmin() ||
        (estaAutenticado() && request.auth.uid == userId &&
         !request.resource.data.diff(resource.data)
           .affectedKeys().hasAny(['aprobado', 'dni', 'cuit', 'email']));

      allow delete: if esAdmin();
    }

    match /usuarios/{userId}/favoritos/{favId} {
      allow read, write: if esAdmin() ||
        (estaAutenticado() && request.auth.uid == userId);
    }

    // ── REPORTES / BUZÓN ─────────────────────────────────────────────────
    match /reportes/{reporteId} {
      allow create: if true;
      allow read, update, delete: if esAdmin();
    }

    // ── ÍNDICE DE LOGIN PÚBLICO (DNI → email) ────────────────────────────
    // Lectura pública por necesidad: se consulta ANTES del login.
    // Los documentos deben contener SOLO { email }, nada más.
    match /login_index/{dni} {
      allow read: if true;
      allow write: if esAdmin();
    }

    // ── CONFIG DEL SISTEMA ───────────────────────────────────────────────
    match /config/{configId} {
        allow read: if true;
        allow write: if esAdmin();
    }

    // ── TODO LO DEMÁS: BLOQUEADO ─────────────────────────────────────────
    match /{document=**} {
      allow read, write: if false;
    }

  }
}
